def convert_time_to_minutes(val):
    if pd.isna(val):
        return np.nan

    val = str(val).strip()
    val = re.sub(r"\.0+$", "", val)

    if ":" in val:
        parts = val.split(":")
        try:
            parts = [float(p) for p in parts]
            if len(parts) == 2:
                return parts[0] * 60 + parts[1]
            elif len(parts) == 3:
                return parts[0] * 60 + parts[1] + parts[2] / 60
        except Exception:
            return np.nan

    try:
        num = float(val)
        if num <= 24:
            return num * 60
        elif num > 1000:
            return num / 60
        else:
            return num
    except Exception:
        return np.nan

def safe_mode(s):
    s = pd.Series(s).dropna()
    return s.mode().iloc[0] if not s.mode().empty else np.nan

def interp_curve(x, y, grid):
    x = np.asarray(x, dtype=float)
    y = np.asarray(y, dtype=float)
    mask = np.isfinite(x) & np.isfinite(y)
    x = x[mask]
    y = y[mask]

    if len(x) == 0:
        return np.full(len(grid), np.nan, dtype=float)

    order = np.argsort(x)
    x = x[order]
    y = y[order]

    x_unique, idx = np.unique(x, return_index=True)
    y_unique = y[idx]

    out = np.full(len(grid), np.nan, dtype=float)
    if len(x_unique) == 1:
        valid = (grid == x_unique[0])
        out[valid] = y_unique[0]
        return out

    valid = (grid >= x_unique.min()) & (grid <= x_unique.max())
    out[valid] = np.interp(grid[valid], x_unique, y_unique)
    return out

def get_feature_names_from_pipeline(fitted_pipe):
    prep = fitted_pipe.named_steps["prep"]
    try:
        return prep.get_feature_names_out()
    except Exception:
        names = []
        for name, trans, cols in prep.transformers_:
            if name == "remainder":
                continue
            if hasattr(trans, "named_steps"):
                last_step = list(trans.named_steps.values())[-1]
                if hasattr(last_step, "get_feature_names_out"):
                    try:
                        step_names = last_step.get_feature_names_out(cols)
                    except Exception:
                        step_names = last_step.get_feature_names_out()
                    names.extend(step_names)
                else:
                    names.extend(cols)
            else:
                names.extend(cols)
        return np.array(names, dtype=object)

class DropAllMissing(BaseEstimator, TransformerMixin):
    def fit(self, X, y=None):
        X = pd.DataFrame(X)
        self.cols_ = X.columns[X.notna().any()].tolist()
        return self

    def transform(self, X):
        X = pd.DataFrame(X)
        for c in self.cols_:
            if c not in X.columns:
                X[c] = np.nan
        return X[self.cols_]

def log1p_safe(x):
    x = np.asarray(x, dtype=float)
    return np.log1p(np.clip(x, a_min=0, a_max=None))

def is_binary_numeric(series):
    s = pd.to_numeric(series, errors="coerce").dropna()
    if s.empty:
        return False
    uniq = set(pd.unique(s))
    return uniq.issubset({0, 1})

def mean_ci(x, z=1.96):
    x = pd.Series(x).dropna().astype(float)
    if len(x) == 0:
        return np.nan, np.nan, np.nan
    mean = x.mean()
    if len(x) == 1:
        return mean, np.nan, np.nan
    se = x.std(ddof=1) / np.sqrt(len(x))
    return mean, mean - z * se, mean + z * se

def compute_ece(y_true, y_prob, n_bins=10):
    y_true = np.asarray(y_true)
    y_prob = np.asarray(y_prob)

    bins = np.linspace(0, 1, n_bins + 1)
    binids = np.digitize(y_prob, bins, right=False) - 1
    binids = np.clip(binids, 0, n_bins - 1)

    ece = 0.0
    for i in range(n_bins):
        mask = binids == i
        if np.sum(mask) == 0:
            continue
        acc = np.mean(y_true[mask])
        conf = np.mean(y_prob[mask])
        ece += np.abs(acc - conf) * (np.sum(mask) / len(y_true))
    return ece

def decision_curve_analysis(y_true, y_prob, thresholds):
    y_true = np.asarray(y_true)
    y_prob = np.asarray(y_prob)
    n = len(y_true)
    net_benefit = []

    for t in thresholds:
        preds = (y_prob >= t).astype(int)
        tp = np.sum((preds == 1) & (y_true == 1))
        fp = np.sum((preds == 1) & (y_true == 0))
        nb = (tp / n) - (fp / n) * (t / (1 - t))
        net_benefit.append(nb)

    return np.array(net_benefit)

def find_optimal_threshold(y_true, y_prob, metric="f1"):
    y_true = np.asarray(y_true)
    y_prob = np.asarray(y_prob)
    thresholds = np.unique(np.round(np.linspace(0.01, 0.99, 199), 6))

    rows = []
    for t in thresholds:
        pred = (y_prob >= t).astype(int)
        tn, fp, fn, tp = confusion_matrix(y_true, pred, labels=[0, 1]).ravel()

        sensitivity = tp / (tp + fn) if (tp + fn) > 0 else np.nan
        specificity = tn / (tn + fp) if (tn + fp) > 0 else np.nan
        ppv = tp / (tp + fp) if (tp + fp) > 0 else np.nan
        npv = tn / (tn + fn) if (tn + fn) > 0 else np.nan
        f1 = f1_score(y_true, pred, zero_division=0)
        acc = accuracy_score(y_true, pred)

        rows.append({
            "threshold": t,
            "f1": f1,
            "accuracy": acc,
            "sensitivity": sensitivity,
            "specificity": specificity,
            "ppv": ppv,
            "npv": npv,
            "tp": tp,
            "fp": fp,
            "tn": tn,
            "fn": fn
        })

    thr_df = pd.DataFrame(rows)

    if metric.lower() == "accuracy":
        best_idx = thr_df["accuracy"].idxmax()
    else:
        best_idx = thr_df["f1"].idxmax()

    return thr_df, thr_df.loc[best_idx].to_dict()

def bootstrap_pr_auc_diff(y_true, prob_a, prob_b, n_boot=2000, random_state=42):
    rng = np.random.default_rng(random_state)
    y_true = np.asarray(y_true)
    prob_a = np.asarray(prob_a)
    prob_b = np.asarray(prob_b)

    diffs = []
    n = len(y_true)

    for _ in range(n_boot):
        idx = rng.integers(0, n, n)
        y_b = y_true[idx]
        if np.unique(y_b).size < 2:
            continue

        pa = prob_a[idx]
        pb = prob_b[idx]

        try:
            auc_a = average_precision_score(y_b, pa)
            auc_b = average_precision_score(y_b, pb)
            diffs.append(auc_a - auc_b)
        except Exception:
            continue

    diffs = np.asarray(diffs, dtype=float)
    if diffs.size == 0:
        return {
            "PR_AUC_Diff_Mean": np.nan,
            "CI_Low": np.nan,
            "CI_High": np.nan,
            "P_bootstrap_two_sided": np.nan,
            "n_boot_valid": 0
        }

    p_two = 2 * min(np.mean(diffs <= 0), np.mean(diffs >= 0))
    p_two = min(p_two, 1.0)

    return {
        "PR_AUC_Diff_Mean": np.mean(diffs),
        "CI_Low": np.percentile(diffs, 2.5),
        "CI_High": np.percentile(diffs, 97.5),
        "P_bootstrap_two_sided": p_two,
        "n_boot_valid": diffs.size
    }

            tpr_interp[-1] = 1.0

        # PR

        precision_rev = precision[::-1]
        pr_interp = interp_curve(recall_rev, precision_rev, pr_grid)

    plt.plot(roc_grid, mean_roc, linewidth=3, label=f"Mean (AUC={fold_df['ROC_AUC'].mean():.3f})")
    plt.plot([0, 1], [0, 1], "--")

    plt.ylabel("True Positive Rate")
    plt.title(f"{model_name} ROC Curves (5 folds + mean)")

    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{model_name}_ROC_5folds_plus_mean.png"), dpi=300)
    plt.close()

    # PR plot
    plt.figure(figsize=(7, 6))
    for i, arr in enumerate(pr_interps, start=1):
        plt.plot(pr_grid, arr, alpha=0.35, label=f"Fold {i}")
    mean_pr = np.nanmean(np.vstack(pr_interps), axis=0)
    plt.plot(pr_grid, mean_pr, linewidth=3, label=f"Mean (AP={fold_df['PR_AUC'].mean():.3f})")
    base_rate = y.mean()
    plt.axhline(base_rate, linestyle="--", label=f"Prevalence={base_rate:.3f}")
    plt.xlabel("Recall")
    plt.ylabel("Precision")
    plt.title(f"{model_name} PR Curves (5 folds + mean)")
    plt.legend(loc="lower left")
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{model_name}_PR_5folds_plus_mean.png"), dpi=300)
    plt.close()

    # Calibration plot
    plt.figure(figsize=(7, 6))
    for i, arr in enumerate(cal_interps, start=1):
        plt.plot(cal_grid, arr, alpha=0.35, label=f"Fold {i}")
    mean_cal = np.nanmean(np.vstack(cal_interps), axis=0)
    plt.plot(cal_grid, mean_cal, linewidth=3, label="Mean")
    plt.plot([0, 1], [0, 1], "--")
    plt.xlabel("Predicted Probability")
    plt.ylabel("Observed Probability")
    plt.title(f"{model_name} Calibration Curves (5 folds + mean)")
    plt.legend(loc="upper left")
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{model_name}_Calibration_5folds_plus_mean.png"), dpi=300)
    plt.close()

    # Aggregate confusion matrices
    cm_sum = np.sum(np.stack(cms, axis=0), axis=0)
    cm_mean = np.mean(np.stack(cms, axis=0), axis=0)

    plt.figure(figsize=(5, 4))

    plt.ylabel("Actual")
    plt.title(f"{model_name} Confusion Matrix (mean across 5 folds)")
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{model_name}_CM_mean_5folds.png"), dpi=300)
    plt.close()

        "y_true": oof_true,
